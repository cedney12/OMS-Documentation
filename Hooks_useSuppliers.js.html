Hooks/useSuppliers.js

import { useQuery } from '@tanstack/react-query'

import { endpoints } from 'Data/api_data'

/**
 * @category Hooks
 * @module useSuppliers
 */
const useSuppliers = () => {

    const fetchSuppliers = async () => {
        const response = await fetch(endpoints.suppliers.getPage(0, 10000))
        if (!response.ok) { throw new Error("Unable to get suppliers") }
        const data = await response.json()
        return data
    }

    const { status, data: suppliers, error, refetch } = useQuery({
        queryKey: ["suppliers"],
        queryFn: fetchSuppliers,
        staleTime: 24 * 60 * 60 * 1000 // 1 day in ms
    })

    return {
        status,
        error,
        refetch,
        suppliers
    }
}

export default useSuppliers
