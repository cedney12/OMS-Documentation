/* eslint-disable react-hooks/exhaustive-deps */
import { Image, Page, Text, View } from "@react-pdf/renderer"
import { keys, map, range } from "lodash"

import logo from 'Assets/logosolid.png'
import { report_styles } from "Data/report_data"
import { formatDate, formatNonZeroNumber, formatPrice } from "Utils/formatters"

const Br = () => "\n"

/**
 * Single-paged report for the Bill of Lading
 * 
 * @param {object} props
 * @param {Order} props.order
 * @param {Supplier} props.supplier
 * @param {Carrier} props.carrier
 * @param {object} props.facility
 * 
 * @category Components
 * @subcategory Reports
 * @module BillOfLading
 * @component 
 */
const BillOfLading = ({
    order,
    supplier,
    carrier,
    facility,
    inputs
}) => {

    const findInputValue = (label) => inputs.find(i => i.label === label).value ?? 0

    const styles = report_styles["Bill Of Lading"]
    const reportData = {
        loadInfo: {
            "Load Number": order.id,
            "Ship Date": formatDate(order.pick_up_date),
            "Delivery Date": formatDate(order.delivery_date),
            "PO Number": order.po_num
        },
        weight: findInputValue("Weight") || 44000,
        cod_amt: findInputValue("C.O.D. Amount") || 0.00,
        cod_fee: findInputValue("C.O.D. Fee") || "Prepaid",
        declared_value: findInputValue("Declared Value") || 0.00
    }

    return (
        <Page style={styles.page} size="LETTER" dpi={72}>
            <Text style={styles.pageTitle}>Bill Of Lading</Text>
            <View style={styles.header}>
                <Image src={logo} style={styles.logo}></Image>
                <View style={styles.loadInfo}>
                    {
                        map(
                            keys(reportData.loadInfo),
                            (key, idx) => (
                                <View key={idx} style={[styles.loadInfoRow, idx === keys(reportData.loadInfo).length - 1 && styles.noBorderBottom]}>
                                    <Text style={styles.loadInfoLabel}>{key}</Text>
                                    <Text style={styles.loadInfoValue}>{reportData.loadInfo[key]}</Text>
                                </View>
                            )
                        )
                    }
                </View>
            </View>
            <View style={styles.table}>
                <View style={styles.tableRow}>
                    <Text style={[styles.tableHeader, { width: "50%" }]}>Shipper</Text>
                    <Text style={[styles.tableHeader, styles.noBorderRight, { width: "50%" }]}>Consignee</Text>
                </View>
                <View style={[styles.tableRow, styles.noBorderBottom]}>
                    <View style={[styles.tableData, { width: "50%" }]}>
                        <Text>{supplier.name}</Text>
                        <Text>{supplier.street}</Text>
                        <Text>{supplier.city}, {supplier.state} {supplier.zip}</Text>
                        <Text>{supplier.phone}</Text>
                    </View>
                    <View style={[styles.tableData, styles.noBorderRight, { width: "50%" }]}>
                        <Text>{facility.basic.Name}</Text>
                        <Text>{facility.basic.Address}</Text>
                        <Text>{facility.basic.Phone}</Text>
                    </View>
                </View>
            </View>
            <View style={[styles.table, { marginTop: -4 }]}>
                <View style={styles.tableRow}>
                    <Text style={[styles.tableHeader, { width: "50%" }]}>3rd Party Billing</Text>
                    <Text style={[styles.tableHeader, styles.noBorderRight, { width: "50%" }]}>Transportation Company</Text>
                </View>
                <View style={[styles.tableRow, styles.noBorderBottom]}>
                    <View style={[styles.tableData, { width: "50%" }]}></View>
                    <View style={[styles.tableData, styles.noBorderRight, { width: "50%" }]}>
                        <Text>{carrier.name} [DOT: {carrier.dot_num}]{carrier.mc_num && ` [MC: ${carrier.mc_num}]`}</Text>
                        <Text>{carrier.street}</Text>
                        <Text>{carrier.city}, {carrier.state} {carrier.zip}{carrier.country !== "USA" && ` ${carrier.country}`}</Text>
                        <Text>{carrier.phone}</Text>
                    </View>
                </View>
            </View>
            <View style={[styles.table, { marginTop: -4 }]}>
                <View style={styles.tableRow}>
                    <Text style={[styles.tableHeader, { width: "40%" }]}>Description of the goods, marks, exceptions</Text>
                    <Text style={[styles.tableHeader, { width: "12%" }]}>Weight (lbs)</Text>
                    <Text style={[styles.tableHeader, { width: "12%" }]}>Type</Text>
                    <Text style={[styles.tableHeader, { width: "12%" }]}>NMFC</Text>
                    <Text style={[styles.tableHeader, { width: "12%" }]}>HM</Text>
                    <Text style={[styles.tableHeader, styles.noBorderRight, { width: "12%" }]}>Class</Text>
                </View>
                <View style={[styles.tableRow, { textAlign: "center" }]}>
                    <Text style={[styles.tableData, { width: "40%" }]}>{order.commodity}</Text>
                    <Text style={[styles.tableData, { width: "12%" }]}>{formatNonZeroNumber({ value: reportData.weight })}</Text>
                    <Text style={[styles.tableData, { width: "12%" }]}>{order.equipment}</Text>
                    <Text style={[styles.tableData, { width: "12%" }]}></Text>
                    <Text style={[styles.tableData, { width: "12%" }]}>{order.has_hazmat && "X"}</Text>
                    <Text style={[styles.tableData, styles.noBorderRight, { width: "12%" }]}></Text>
                </View>
                {map(range(0, 4), (_, idx) => {
                    return (
                        <View key={idx} style={[styles.tableRow, { textAlign: "center" }]}>
                            <Text style={[styles.tableData, { width: "40%" }]}></Text>
                            <Text style={[styles.tableData, { width: "12%" }]}></Text>
                            <Text style={[styles.tableData, { width: "12%" }]}></Text>
                            <Text style={[styles.tableData, { width: "12%" }]}></Text>
                            <Text style={[styles.tableData, { width: "12%" }]}></Text>
                            <Text style={[styles.tableData, styles.noBorderRight, { width: "12%" }]}></Text>
                        </View>
                    )
                })}
                <View style={[styles.tableRow, styles.noBorderBottom, { textAlign: "center" }]}>
                    <Text style={[styles.tableData, { width: "40%" }]}></Text>
                    <Text style={[styles.tableData, { width: "12%" }]}>Total Weight<Br />{formatNonZeroNumber({ value: reportData.weight })}</Text>
                    <Text style={[styles.tableData, styles.noBorderRight, { width: "48%", paddingTop: 1 }]}>Emergency Response Phone</Text>
                </View>
            </View>
            <View style={[styles.table, { marginTop: -4, flexDirection: "row" }]}>
                <View style={[styles.tableData, { width: "65%" }]}>
                    <Text>Notes:</Text>
                    <Text>{order.notes}</Text>
                </View>
                <View style={[{ width: "35%", textAlign: "center" }]}>
                    <Text style={{ borderBottom: "0.75pt solid black", padding: 2 }}>C.O.D. Amount: ${formatPrice(reportData.cod_amt)}</Text>
                    <Text style={{ borderBottom: "0.75pt solid black", padding: 2 }}>C.O.D. Fee: {reportData.cod_fee}</Text>
                    <Text style={{ borderBottom: "0.75pt solid black", padding: 2 }}>Declared Value: ${formatPrice(reportData.declared_value)}</Text>
                    <Text style={{ fontSize: 8, paddingBottom: 16 }}>If at cosignor's risk, write or stamp here</Text>
                </View>
            </View>
            <View style={[styles.fillInTable, { marginTop: -4 }]}>
                <View style={styles.fillInColumn}>
                    <Text style={styles.fillInCell}>Shipper</Text>
                    <Text style={[styles.fillInCell, styles.noBorderBottom]}>Per</Text>
                </View>
                <View style={styles.fillInColumn}>
                    <Text style={styles.fillInCell}>Carrier</Text>
                    <Text style={[styles.fillInCell, styles.noBorderBottom]}>Per</Text>
                </View>
                <View style={styles.fillInColumn}>
                    <Text style={styles.fillInCell}>Date</Text>
                    <Text style={[styles.fillInCell, styles.noBorderBottom]}>Time</Text>
                </View>
                <View style={[styles.fillInColumn, { borderRight: "none" }]}>
                    <Text style={[styles.fillInCell, styles.noBorderBottom]}>Number of Pieces Received</Text>
                </View>
            </View>
            <View style={[styles.fillInTable, { marginTop: -4 }]}>
                <View style={styles.fillInColumn}>
                    <Text style={[styles.fillInCell, styles.noBorderBottom, { paddingBottom: 16 }]}>Consignee Name</Text>
                </View>
                <View style={styles.fillInColumn}>
                    <Text style={[styles.fillInCell, styles.noBorderBottom, { paddingBottom: 16 }]}>Date</Text>
                </View>
                <View style={styles.fillInColumn}>
                    <Text style={[styles.fillInCell, styles.noBorderBottom, { paddingBottom: 16 }]}>Signature</Text>
                </View>
                <View style={[styles.fillInColumn, { borderRight: "none" }]}>
                    <Text style={[styles.fillInCell, styles.noBorderBottom, { paddingBottom: 16 }]}>Number of Pieces Received</Text>
                </View>
            </View>
        </Page>
    )
}

export default BillOfLading
