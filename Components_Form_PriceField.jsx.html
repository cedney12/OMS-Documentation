import { TextField } from "@mui/material"
import React, { useState } from "react"
import { NumericFormat } from "react-number-format"

/**
 * Formatted Input for USD Price
 * 
 * Formats value to 2 decimals, and takes any generic input props along with
 * the named props (such as label, placeholder, name). To parse number, use `parsePriceFieldValue`
 * 
 * Can function as controlled or uncontrolled input
 * 
 * @param {object} props
 * @param {string} [props.value] Controlled value
 * @param {React.SetStateAction<React.Dispatch<string>>} [props.setValue] Function to set controlled value
 * @param {string} [props.defaultValue] Uncontrolled value
 * @param {object} props.sx Inline CSS rules
 * 
 * @returns {JSX.Element}
 * @component
 * @category Components
 * @subcategory Form
 * @module PriceField
 */
const PriceField = ({
    value,
    setValue,
    defaultValue,
    sx,
    width,
    ...rest
}) => {

    sx = {
        ...sx,
        width
    }

    const [localValue, setLocalValue] = useState(defaultValue ?? value)

    const handleChange = (values) => {
        setValue && setValue(values.value)
        setLocalValue(values.value)
    }

    return (
        <NumericFormat
            placeholder="$123.00"
            {...rest}
            sx={sx}
            prefix="$"
            customInput={TextField}
            value={localValue}
            onValueChange={handleChange}
            decimalScale={2}
            fixedDecimalScale
            allowNegative={false}
            thousandSeparator=","
        />
    )
}

export default PriceField
