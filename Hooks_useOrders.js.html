Hooks/useOrders.js

import { useQuery } from "@tanstack/react-query"
import { useEffect, useState } from "react"
import { useLocation, useSearchParams } from "react-router-dom"

import { endpoints } from "Data/api_data"

/**
 * Custom hook that streamlines the fetching of order data, caching it 
 * based on facility, booked status, pagination, sorting, and search value.
 * 
 * @param {string} facility Facility to filter orders in
 * @param {boolean} isBooked `true` if looking for booked orders, `false` otherwise
 * @category Hooks
 * @module useOrders
 */
const useOrders = (facility, isBooked) => {

    const location = useLocation()

    const [paginationModel, setPaginationModel] = useState(() => {
        const params = new URLSearchParams(location.search)

        return {
            page: (Number(params.get("page")) || 1) - 1,
            pageSize: Number(params.get("pageSize")) || 50
        }
    })
    const [sortModel, setSortModel] = useState(() => {
        const params = new URLSearchParams(location.search)

        return {
            sortField: params.get("sortField") || null,
            sortDirection: params.get("sortDirection") || null
        }
    })
    const [searchValue, setSearchValue] = useState(() => {
        const params = new URLSearchParams(location.search)

        return params.get("search") || ""
    })
    const [searchParams, setSearchParams] = useSearchParams()

    useEffect(() => {
        const updateQueryParams = (params) => {
            const newParams = new URLSearchParams(searchParams)
            for (const pair of params) {
                if (pair.value === "" || pair.value == null) {
                    newParams.delete(pair.key)
                    continue
                }

                newParams.set(pair.key, pair.value)
            }
            setSearchParams(newParams)
        }

        updateQueryParams([
            { key: "page", value: paginationModel.page + 1 },
            { key: "pageSize", value: paginationModel.pageSize },
            { key: "sortField", value: sortModel.sortField },
            { key: "sortDirection", value: sortModel.sortDirection },
            { key: "search", value: searchValue },
        ])
    }, [paginationModel, sortModel, searchValue])

    const fetchOrders = async ({ queryKey }) => {
        const { facility, paginationModel, sortModel, isBooked, search } = queryKey[1]
        try {
            const response = await fetch(
                endpoints.orders.getPage(
                    paginationModel.page,
                    paginationModel.pageSize,
                    isBooked,
                    facility,
                    search,
                    sortModel.sortField,
                    sortModel.sortDirection
                )
            )
            const data = await response.json()
            return data
        } catch (e) {
            console.error(e)
            return []
        }
    }

    const fetchCount = async ({ queryKey }) => {
        const { facility, isBooked } = queryKey[1]
        try {
            const response = await fetch(
                endpoints.orders.getCount(
                    isBooked,
                    facility,
                )
            )
            const data = await response.json()
            return data
        } catch (e) {
            console.error(e)
            return null
        }
    }

    const dataQuery = useQuery({
        queryKey: ['orders', { facility, paginationModel, sortModel, isBooked, search: searchValue }],
        queryFn: fetchOrders,
        staleTime: 5 * 60 * 1000,
        cacheTime: 10 * 60 * 1000,
        keepPreviousData: true,
    })

    const countQuery = useQuery({
        queryKey: ['count', { facility, isBooked }],
        queryFn: fetchCount,
        staleTime: 5 * 60 * 1000,
        cacheTime: 10 * 60 * 1000,
        keepPreviousData: true,
    })

    return {
        dataQuery,
        countQuery,
        controls: {
            paginationModel,
            setPaginationModel,
            sortModel,
            setSortModel,
            searchValue,
            setSearchValue
        }
    }
}

export default useOrders
