Components/Modals/CarrierModal.jsx

import { Button, Checkbox, FormControlLabel, TextField } from "@mui/material"
import { capitalize, isNaN, mapKeys, mapValues } from "lodash"
import { enqueueSnackbar } from "notistack"

import Autocomplete from "Components/Form/Autocomplete"
import PhoneField from "Components/Form/PhoneField"
import { state_abbreviations } from "Data/form_data"

import styles from './Modals.module.css'
import ModalWrapper from "./ModalWrapper"
import { DatePicker } from "@mui/x-date-pickers"
import dayjs from "dayjs"
import { useState } from "react"
import { convertFormData } from "Utils/functions"
import useAPIMutation from "Hooks/useAPIMutation"
import { endpoints } from "Data/api_data"

/**
 * Modal for creating and editing Carriers
 * 
 * @param {object} props
 * @param {boolean} props.open `true` if the modal is open, `false` otherwise
 * @param {function} props.onClose Function that occurs when the modal closes
 * @param {"add" | "edit"} props.type Type of modal being displayed, either creation or editing
 * @param {Carrier} props.carrier Carrier to edit, if editing
 * @param {function} props.reloadData Function to reload API data after creation/editing
 * 
 * @category Components
 * @subcategory Modals
 * @module CarrierModal
 * @component
 */
const CarrierModal = ({
    open,
    onClose,
    type,
    carrier = {},
}) => {
    const { mutate } = useAPIMutation(
        () => {
            enqueueSnackbar(`Successfully ${type}ed carrier`)
            onClose()
        },
        () => enqueueSnackbar(`Unable to ${type} carrier`)
    )

    const [coi, setCOI] = useState(carrier.coi || false)

    const handleSubmit = async (e) => {
        e.preventDefault()

        if (!e.target.phone.value.match(/\(\d{3}\) \d{3}-\d{3}/g)) {
            enqueueSnackbar("Please input a valid phone number", { variant: "error" })
            return
        }

        let data = mapValues({
            ...carrier,
            ...Object.fromEntries(new FormData(e.target))
        }, convertFormData)
        data.country = state_abbreviations.find(s => s.state === data.state)?.country ?? "USA"
        data.zip = `${data.zip}`
        data.mc_num = `${data.mc_num}`

        const url = type === "add" ? endpoints.carriers.post : endpoints.carriers.put(data.id)
        const method = type === "add" ? "POST" : "PUT"

        mutate({ url, method, data })
    }

    return (
        <ModalWrapper open={open} onClose={onClose} sx={{ width: "750px" }}>
            <h1>{capitalize(type)} Carrier</h1>
            <form onSubmit={handleSubmit}>
                <div className={styles.poCards}>
                    <div className={styles.formCard}>
                        <h2>Carrier</h2>
                        <TextField
                            label="Carrier"
                            name="name"
                            defaultValue={carrier.name ?? ""}
                            sx={{ width: "328px" }}
                            required
                        />
                        <div className={styles.formRow}>
                            <TextField
                                label="DOT #"
                                name="dot_num"
                                defaultValue={carrier.dot_num ?? ""}
                                sx={{ width: "160px" }}
                                required
                            />
                            <TextField
                                label="MC #"
                                name="mc_num"
                                defaultValue={carrier.mc_num ?? ""}
                                sx={{ width: "160px" }}
                            />
                        </div>
                    </div>
                    <div className={styles.formCard}>
                        <h2>Contact Info</h2>
                        <TextField
                            label="Rate Quote"
                            name="rate_quote"
                            defaultValue={carrier.rate_quote ?? ""}
                            sx={{ width: "328px" }}
                        />
                        <TextField
                            label="Email"
                            name="email"
                            defaultValue={carrier.email ?? ""}
                            sx={{ width: "328px" }}
                            required
                        />
                        <PhoneField
                            name="phone"
                            defaultValue={carrier.phone ?? ""}
                            required
                        />
                    </div>
                    <div className={styles.formCard}>
                        <h2>Location Info</h2>
                        <TextField
                            label="Main Hub Location"
                            name="hub_location"
                            defaultValue={carrier.hub_location ?? ""}
                            sx={{ width: "367px" }}
                            required
                        />
                        <TextField
                            label="Street"
                            name="street"
                            defaultValue={carrier.street ?? ""}
                            sx={{ width: "367px" }}
                            required
                        />
                        <div className={styles.formRow}>
                            <TextField
                                label="City"
                                name="city"
                                defaultValue={carrier.city ?? ""}
                                sx={{ width: "150px" }}
                                required
                            />
                            <Autocomplete
                                label="ST"
                                name="state"
                                width="100px"
                                options={state_abbreviations}
                                defaultValue={carrier.state ? { state: carrier.state } : ""}
                                groupBy={(option) => option.country}
                                getOptionLabel={(option) => option.state ?? ""}
                                isOptionEqualToValue={(option, value) => option.state === value.state}
                            />
                            <TextField
                                label="ZIP"
                                name="zip"
                                defaultValue={carrier.zip ?? ""}
                                sx={{ width: "100px" }}
                                required
                            />
                        </div>
                    </div>

                    <div className={styles.formCard}>
                        <h2>Carrier Details</h2>
                        <div className={styles.formRow}>
                            <TextField
                                label="Drivers"
                                name="drivers"
                                defaultValue={carrier.drivers ?? ""}
                                sx={{ width: "140px" }}
                                type="number"
                            />
                            <TextField
                                label="Trucks"
                                name="trucks"
                                defaultValue={carrier.trucks ?? ""}
                                sx={{ width: "140px" }}
                                type="number"
                            />
                        </div>
                        <div className={styles.formRow}>
                            <TextField
                                label="Trailers"
                                name="trailers"
                                defaultValue={carrier.trailers ?? ""}
                                sx={{ width: "140px" }}
                                type="number"
                            />
                            <TextField
                                label="Flatbeds"
                                name="flatbeds"
                                defaultValue={carrier.flatbeds ?? ""}
                                sx={{ width: "140px" }}
                                type="number"
                            />
                        </div>
                        <div className={styles.formRow} style={{ alignItems: "start" }}>
                            <div style={{
                                display: "flex",
                                flexDirection: "column",
                                gap: "8px"
                            }}>
                                <TextField
                                    label="Dumps"
                                    name="dumps"
                                    defaultValue={carrier.dumps ?? ""}
                                    sx={{ width: "140px" }}
                                    type="number"
                                />
                                <DatePicker
                                    label="COI Expires"
                                    name="coi_expire_date"
                                    defaultValue={carrier.coi_expire_date && dayjs(carrier.coi_expire_date)}
                                    sx={{ width: "140px" }}
                                    required
                                    disabled={!coi}
                                />
                            </div>
                            <div style={{
                                display: "flex",
                                flexDirection: "column",
                                gap: "4px"
                            }}>
                                <FormControlLabel
                                    control={
                                        <Checkbox
                                            defaultChecked={carrier.map}
                                            sx={{ marginRight: "4px", padding: 0 }}
                                        />
                                    }
                                    label="Map"
                                    name="map"
                                />
                                <FormControlLabel
                                    control={
                                        <Checkbox
                                            defaultChecked={carrier.ships_hazmat}
                                            sx={{ marginRight: "4px", padding: 0 }}
                                        />
                                    }
                                    label="Haz"
                                    name="ships_hazmat"
                                />
                                <FormControlLabel
                                    control={
                                        <Checkbox
                                            defaultChecked={carrier.mca}
                                            sx={{ marginRight: "4px", padding: 0 }}
                                        />
                                    }
                                    label="MCA"
                                    name="mca"
                                />
                                <FormControlLabel
                                    control={
                                        <Checkbox
                                            checked={coi}
                                            onChange={(_, value) => setCOI(value)}
                                            sx={{ marginRight: "4px", padding: 0 }}
                                        />
                                    }
                                    label="COI"
                                    name="coi"
                                />
                            </div>
                        </div>
                    </div>
                </div>
                <Button type="submit">{type}</Button>
            </form>
        </ModalWrapper>
    )
}

export default CarrierModal
