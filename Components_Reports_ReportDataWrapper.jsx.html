Components/Reports/ReportDataWrapper.jsx

import { Document, PDFViewer } from "@react-pdf/renderer"
import { find, isEmpty } from "lodash"
import { useEffect, useState } from "react"

import { useAuth } from "Contexts/AuthContext"
import { useLoading } from "Contexts/LoadingContext"
import { endpoints } from "Data/api_data"
import { getFacilityData } from "Data/facility_data"
import useQueryData from "Hooks/useQueryData"
import { areAnyEmpty } from "Utils/functions"

import BillOfLading from "./BillOfLading"
import RateConfirmation from "./RateConfirmation"

/**
 * Wrapper to send data to the correct reports
 * 
 * This was created to separate the PDF creation and the data fetching processes.
 * When a component is inside of react-pdf/renderer's `PDFViewer`, no contexts can be accessed.
 * Thus, useQuery, useAuth, and useLoading are unavailable, which are used for data fetching.
 * This component is used to get around that, as well as make it easier for combined reports.
 * 
 * @param {object} props
 * 
 * @category Components
 * @subcategory Reports
 * @module ReportDataWrapper
 * @component
 */
const ReportDataWrapper = ({
    reportLabel,
    orderID,
    updateInstance,
    inputs
}) => {
    const { user } = useAuth()
    const { startLoading, stopLoading } = useLoading()
    const { data: order, isFetched: isOrderFetched, error: orderError } = useQueryData(endpoints.orders.getByID(orderID))
    const { data: supplier, isFetched: isSupplierFetched, error: supplierError, setUrl: setSupplierUrl } = useQueryData()
    const { data: carrier, isFetched: isCarrierFetched, error: carrierError, setUrl: setCarrierUrl } = useQueryData()
    const { data: contacts, isFetched: isContactsFetched, error: contactsError, setUrl: setContactsUrl } = useQueryData()

    const [facility, setFacility] = useState()
    const [doc, setDoc] = useState()
    const [error, setError] = useState()

    useEffect(startLoading, [])

    useEffect(() => {
        if (isEmpty(order)) return

        setSupplierUrl(endpoints.suppliers.getSupplierByName(order.supplier))
        setCarrierUrl(endpoints.carriers.getByName(order.carrier))
        setContactsUrl(endpoints.contacts.getSupplierContactsByName(order.supplier))
        setFacility(getFacilityData(order))
    }, [order])

    useEffect(() => {
        const error = [
            orderError,
            supplierError,
            carrierError,
            contactsError
        ].find(error => error)

        if (error) {
            setError(`Error loading PDF: ${error.message}`)
            stopLoading()
        }
    }, [orderError, supplierError, carrierError, contactsError])

    useEffect(() => {
        const emptyFetch = [
            [isOrderFetched, order, "order"],
            [isSupplierFetched, supplier, "supplier"],
            [isCarrierFetched, carrier, "carrier"],
            [isContactsFetched, contacts, "contacts"]
        ].find(([isFetched, data]) => isFetched && isEmpty(data))

        if (emptyFetch) {
            setError(`Error loading PDF: No data for ${emptyFetch[2]}`)
            stopLoading()
        }
    }, [isOrderFetched, isSupplierFetched, isCarrierFetched, isContactsFetched])

    useEffect(() => {
        startLoading()
        let pages = null

        if (error) {
            stopLoading()
            return
        }

        if (areAnyEmpty(order, supplier, carrier, contacts)) return

        switch (reportLabel) {
            case "Bill Of Lading":
                pages = (
                    <BillOfLading
                        order={order}
                        supplier={supplier}
                        carrier={carrier}
                        facility={facility}
                        inputs={inputs}
                    />
                )
                break
            case "Rate Confirmation":
                pages = (
                    <RateConfirmation
                        order={order}
                        supplier={supplier}
                        carrier={carrier}
                        contact={find(contacts, (c) => c.id === supplier.primary_contact_id) ?? contacts[0]}
                        facility={facility}
                        user={user}
                        inputs={inputs}
                    />
                )
                break
            case "Bill & Confirmation":
                pages = (
                    <>
                        <BillOfLading
                            order={order}
                            supplier={supplier}
                            carrier={carrier}
                            facility={facility}
                            inputs={inputs}
                        />
                        <RateConfirmation
                            order={order}
                            supplier={supplier}
                            carrier={carrier}
                            contact={find(contacts, (c) => c.id === supplier.primary_contact_id) ?? contacts[0]}
                            facility={facility}
                            user={user}
                            inputs={inputs}
                        />
                    </>
                )
                break
        }

        setDoc(<Document>{pages}</Document>)
        updateInstance(<Document>{pages}</Document>)
    }, [reportLabel, order, supplier, carrier, contacts, error])

    return (
        error
            ? <h1>{error}</h1>
            : doc
                ? <PDFViewer width="100%" height="100%" onLoad={stopLoading}>{doc}</PDFViewer>
                : <></>
    )
}

export default ReportDataWrapper
