Components/Modals/POModal.jsx

import { useMsal } from "@azure/msal-react"
import { Button, capitalize, Checkbox, FormControl, FormControlLabel, FormLabel, Radio, RadioGroup, TextField } from "@mui/material"
import { DatePicker } from "@mui/x-date-pickers"
import dayjs from "dayjs"
import { isEmpty, mapValues } from "lodash"
import { enqueueSnackbar } from "notistack"
import { useState } from "react"

import Autocomplete from "Components/Form/Autocomplete"
import PriceField from "Components/Form/PriceField"
import TextArea from "Components/Form/TextArea"
import { endpoints } from "Data/api_data"
import { commodities, facilities, redCommodities, statuses, truckTypes } from "Data/order_data"
import useCarriers from "Hooks/useCarriers"
import useAPIMutation from "Hooks/useAPIMutation"
import useSuppliers from "Hooks/useSuppliers"
import useTraders from "Hooks/useTraders"
import { convertFormData, getInitials } from "Utils/functions"

import styles from './Modals.module.css'
import ModalWrapper from "./ModalWrapper"

/**
 * Modal to allow for the creation and editing of Orders
 * 
 * @param {object} props
 * @param {boolean} props.open `true` if the modal is open, `false` otherwise
 * @param {function} props.onClose Function that occurs when the modal closes
 * @param {"create" | "edit"} props.type Type of modal open, either creating or editing
 * @param {Order} [props.order] Order to edit, if editing
 * @param {React.Dispatch<React.SetStateAction<number>>} props.setUpdatedRowId Function to set the row most recently updated 
 * @param {function} props.reloadData Function to reload API data on POST/PUT
 * 
 * @category Components
 * @subcategory Modals
 * @module POModal
 * @component
 */
const POModal = ({
    open,
    onClose,
    type,
    order = {},
    setUpdatedRowId,
    onSuccess
}) => {
    const { accounts } = useMsal()
    const { mutate, mutateAsync } = useAPIMutation(
        (data) => {
            type === "create" && onSuccess()
            setUpdatedRowId(data.id)
            enqueueSnackbar("Successfully created order", { variant: "success" })
            onClose()
        },
        () => enqueueSnackbar("Unable to create order", { variant: "error" }),
    )

    const handleSubmit = async (e) => {
        e.preventDefault()
        if (hazmat == null) {
            enqueueSnackbar("You MUST declare if the load contains batteries or hazardous materials", { variant: "error" })
            return
        }

        let suffix = ""
        if (type === "create") {
            const previousOrders = await mutateAsync({
                url: endpoints.orders.getPage(0, 1, null, null, e.target.po_num.value, "id", "desc"),
                method: "GET"
            })
            if (!isEmpty(previousOrders)) {
                const lastLoad = previousOrders[0]
                const suffixIndex = lastLoad.po_num.search(/-\d+/g)
                if (suffixIndex === -1) {
                    await handleRequest(
                        endpoints.orders.putOne(lastLoad.id),
                        "PUT",
                        {
                            ...lastLoad,
                            po_num: lastLoad.po_num + "-1"
                        }
                    )
                    suffix = "-2"
                }
                else {
                    const lastLoadSuffix = Number(lastLoad.po_num.substring(suffixIndex + 1))
                    if (lastLoadSuffix) suffix = `-${lastLoadSuffix + 1}`
                }
                if (!window.confirm(`Purchase Order with number ${e.target.po_num.value} already exists, create another load with PO # ${e.target.po_num.value}${suffix}`)) return
            }
        }

        const data = mapValues({
            ...order,
            ...Object.fromEntries(new FormData(e.target)),
            has_hazmat: hazmat === "Yes",
        }, convertFormData)
        data.po_num = data.po_num + suffix
        data.status_initials = ["Booked", "Scheduled"].includes(data.status) ? getInitials(accounts[0].name) : null
        data.pick_up_date = cd || !data.pick_up_date ? null : data.pick_up_date
        data.delivery_date = !data.delivery_date ? null : data.delivery_date
        data.carrier = cd ? null : data.carrier
        data.equipment = cd ? null : data.equipment
        data.rate = cd ? null : data.rate
        data.pickup_location = cd ? null : data.pickup_location
        data.hours = cd ? null : data.hours

        const url = type === "create"
            ? endpoints.orders.post
            : endpoints.orders.putOne(order.id)
        const method = type === "create" ? "POST" : "PUT"

        mutate({ url, method, data })
    }

    const [cd, setCD] = useState(order.customer_delivery)
    const [hazmat, setHazmat] = useState(order.has_hazmat != null ? order.has_hazmat ? "Yes" : "No" : null)

    const { traders } = useTraders()
    const { suppliers } = useSuppliers()
    const { carriers } = useCarriers()

    return (
        <ModalWrapper
            open={open}
            onClose={onClose}
            sx={{ width: "825px" }}
        >
            <h1>{capitalize(type)} a Purchase Order</h1>
            <form onSubmit={handleSubmit}>
                <div className={styles.poCards}>
                    <div className={styles.formCard}>
                        <h2>Order Info</h2>
                        <div className={styles.formRow}>
                            <TextField
                                label="PO #"
                                name="po_num"
                                defaultValue={order.po_num}
                                sx={{ width: "25%" }}
                                required
                            />
                            <Autocomplete
                                width="75%"
                                options={traders}
                                label="Trader"
                                name="trader"
                                defaultValue={order.trader}
                                required
                            />
                        </div>
                        <div className={styles.formRow}>
                            <Autocomplete
                                width="50%"
                                options={facilities}
                                label="Facility"
                                name="facility"
                                defaultValue={order.facility}
                                required
                            />
                            <Autocomplete
                                width="50%"
                                options={statuses}
                                label="Status"
                                name="status"
                                defaultValue={{ value: order.status } || ""}
                                getOptionLabel={(option) => option.value || ""}
                                isOptionEqualToValue={(option, value) => option.value === value.value}
                                required
                            />
                        </div>
                        <Autocomplete
                            width="381px"
                            options={commodities}
                            label="Commodity"
                            name="commodity"
                            defaultValue={order.commodity}
                            renderOption={(props, option) => {
                                const { key, ...optionProps } = props
                                const color = redCommodities.includes(option) ? "#f60000" : "black"
                                return (
                                    <li key={key} {...optionProps} style={{ color }}>
                                        {option}
                                    </li>
                                )
                            }}
                            required
                        />
                        <Autocomplete
                            width="381px"
                            label="Supplier"
                            name="supplier"
                            defaultValue={{ name: order.supplier } || ""}
                            options={suppliers}
                            getOptionLabel={(option) => option.name || ""}
                            isOptionEqualToValue={(option, value) => option.name === value.name}
                            required
                        />
                    </div>
                    <div className={styles.formCard}>
                        <div className={styles.formRow} style={{ justifyContent: "space-between" }}>
                            <h2>Carrier Info</h2>
                            <FormControlLabel
                                control={
                                    <Checkbox
                                        checked={cd ?? false}
                                        onChange={(_, v) => setCD(v)}
                                        sx={{ marginLeft: "4px", padding: 0 }}
                                    />
                                }
                                label="Customer Delivery"
                                labelPlacement="start"
                                name="customer_delivery"
                            />
                        </div>
                        <Autocomplete
                            width="350px"
                            label="Carrier"
                            name="carrier"
                            defaultValue={{ name: order.carrier } || ""}
                            options={carriers}
                            getOptionLabel={(option) => option.name || ""}
                            isOptionEqualToValue={(option, value) => option.name === value.name}
                            disabled={cd}
                            required
                        />
                        <div className={styles.formRow}>
                            <Autocomplete
                                width="50%"
                                label="Equipment"
                                name="equipment"
                                defaultValue={order.equipment}
                                options={truckTypes}
                                disabled={cd}
                                required
                            />
                            <PriceField
                                label="Rate"
                                name="rate"
                                defaultValue={order.rate}
                                disabled={cd}
                                width="50%"
                                required
                            />
                        </div>
                        <TextField
                            label="Pickup Location"
                            name="pickup_location"
                            defaultValue={order.pickup_location}
                            required
                            disabled={cd}
                        />
                        <TextField
                            label="Hours"
                            name="hours"
                            defaultValue={order.hours}
                            disabled={cd}
                        />
                    </div>
                    <div className={styles.formCard}>
                        <h2>Order Details</h2>
                        <div className={styles.formRow}>
                            <DatePicker
                                label="Date Issued"
                                name="date_issued"
                                defaultValue={dayjs(order.date_issued ?? new Date())}
                            />
                            <DatePicker
                                label="Pick Up Date"
                                name="pick_up_date"
                                defaultValue={
                                    order.pick_up_date
                                    && dayjs(order.pick_up_date)
                                }
                                disabled={cd}
                            />
                            <DatePicker
                                label="Delivery Date"
                                name="delivery_date"
                                defaultValue={
                                    order.delivery_date
                                    && dayjs(order.delivery_date)
                                }
                            />
                            <TextField
                                sx={{ width: "281px" }}
                                label="GPS Link"
                                name="gps_link"
                                defaultValue={order.gps_link}
                            />
                        </div>
                        <div className={styles.formRow} style={{ justifyContent: "space-between" }}>
                            <FormControlLabel
                                control={
                                    <Checkbox
                                        defaultChecked={order.bill_received}
                                        sx={{ marginRight: "4px", padding: 0 }}
                                    />
                                }
                                label="Bill of Lading"
                                name="bill_received"
                            />
                            <FormControl style={{ display: "flex", gap: "8px", flexDirection: "row", alignItems: "center" }}>
                                <p><b>Load contains batteries or hazardous materials?</b></p>
                                <RadioGroup
                                    row
                                    sx={{ gap: "8px" }}
                                    value={hazmat}
                                    onChange={(_, value) => setHazmat(value)}
                                >
                                    <FormControlLabel value="Yes" control={<Radio />} label="Yes" />
                                    <FormControlLabel value="No" control={<Radio />} label="No" />
                                </RadioGroup>
                            </FormControl>
                        </div>
                        <TextArea
                            label="Notes"
                            name="notes"
                            defaultValue={order.notes}
                        />
                    </div>
                </div>
                <Button type="submit">{capitalize(type)}</Button>
            </form>
        </ModalWrapper>
    )
}

export default POModal
