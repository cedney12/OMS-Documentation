Components/Buttons/ReportDropdown.jsx

import { useEffect, useState } from "react"

import styles from './Buttons.module.css'
import { GenerateReportButton } from "./IconButtons"

/**
 * Dropdown menu for report options, so that selecting a report to generate
 * is not difficult or distracting
 * 
 * Currently has the following options:
 * - Bill Of Lading
 * - Rate & Load Confirmation
 * - Both Bill & Confirmation
 * 
 * @category Components
 * @subcategory Buttons
 * @module ReportDropdown
 * @component
 */
const ReportDropdown = ({
    openModal,
    disabled
}) => {
    const [open, setOpen] = useState(false)

    useEffect(() => {
        setOpen(prev => !disabled ? false : prev)
    }, [disabled])

    const handleClick = (reportLabel) => {
        openModal(reportLabel)
        setOpen(false)
    }

    return (
        <div className={styles.dropdown}>
            <GenerateReportButton
                disabled={disabled}
                onClick={() => setOpen(prev => !prev)}
            />
            {
                open &&
                <div className={styles.options}>
                    <div className={styles.arrow}>
                        <div></div>
                    </div>
                    <h2 onClick={() => handleClick("Bill Of Lading")}>Bill Of Lading</h2>
                    <h2 onClick={() => handleClick("Rate Confirmation")}>Rate & Load Confirmation</h2>
                    <h2 onClick={() => handleClick("Bill & Confirmation")}>Both</h2>
                </div>
            }
        </div>
    )
}

export default ReportDropdown
