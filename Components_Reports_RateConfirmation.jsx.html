/* eslint-disable react-hooks/exhaustive-deps */
import { Image, Page, Text, View } from "@react-pdf/renderer"
import { keys } from "lodash"

import logo from 'Assets/logosolid.png'
import { report_styles } from "Data/report_data"
import { formatDate, formatNonZeroNumber, formatPrice } from "Utils/formatters"

/**
 * 
 * @param {object} props
 * @param {Order} props.order
 * @param {Supplier} props.supplier
 * @param {Carrier} props.carrier
 * @param {Contact} props.contact
 * @param {object} props.facility
 * @param {MSALUser} props.user
 * 
 * @category Components
 * @subcategory Reports
 * @module RateConfirmation
 * @component
 */
const RateConfirmation = ({
    order,
    supplier,
    carrier,
    contact,
    facility,
    user,
    inputs
}) => {
    const styles = report_styles["Rate Confirmation"]

    const findInputValue = (label) => inputs.find(i => i.label === label).value ?? 0

    const reportData = {
        loadInfo: {
            "Logistics Specialist": user.displayName,
            "Phone #": (user.businessPhones?.length > 0 && user.businessPhones[0]) || "(877) 200-0840",
            "Email": user.mail,
            "Load #": order.id,
            "Ship Date": formatDate(order.delivery_date),
            "Today's Date": formatDate(new Date().toJSON())
        },
        carrier: {
            "Carrier": {
                value: carrier.name,
                width: "25%"
            },
            "DOT #": {
                value: carrier.dot_num,
                width: "10%"
            },
            "MC #": {
                value: carrier.mc_num,
                width: "10%"
            },
            "Phone #": {
                value: carrier.phone,
                width: "20%"
            },
            "Equipment": {
                value: order.equipment,
                width: "10%"
            },
            "Agreed Amount": {
                value: `$${formatPrice(order.rate)}`,
                width: "15%"
            },
            "Status": {
                value: order.status,
                width: "10%"
            }
        },
        shipper: {
            basic: {
                "Name": supplier.name,
                "Address": `${supplier.street}\n${supplier.city}, ${supplier.state} ${supplier.zip}`,
                "Phone": supplier.phone,
                "Contact": `${contact.first_name} ${contact.last_name}`,
            },
            details: [{
                "Date": formatDate(order.pick_up_date),
                "Type": order.equipment,
                "Weight": `${formatNonZeroNumber({ value: findInputValue("Weight") || 44000 })} lbs`,
                "Notes": supplier.notes
            }, {
                "Purchase Order #": order.po_num,
                "Shipping Hours": order.hours,
                "Commodity": order.commodity
            }],
            notes: order.notes
        },
        consignee: {
            ...facility
        },
    }

    return (
        <Page style={styles.page} size="LETTER" dpi={72}>
            <Text style={styles.pageTitle}>Rate Confirmation</Text>
            <View style={styles.header}>
                <Image src={logo} style={styles.logo}></Image>
                <View style={styles.loadInfo}>
                    {
                        keys(reportData.loadInfo).map((key, idx) => (
                            <View
                                key={`info_${idx}`}
                                style={[styles.loadInfoRow, {
                                    borderTop: idx === 0
                                        ? "none"
                                        : "0.75pt solid black"
                                }]}
                            >
                                <Text style={styles.loadInfoLabel}>{key}</Text>
                                <Text style={styles.loadInfoValue}>{reportData.loadInfo[key]}</Text>
                            </View>
                        ))
                    }
                </View>
            </View>

            <View style={styles.table}>
                {
                    keys(reportData.carrier).map((key, idx) => (
                        <View
                            key={`carrier_${idx}`}
                            style={[styles.carrierTableColumn, {
                                width: reportData.carrier[key].width,
                                borderLeft: idx === 0 ? "none" : "0.75pt solid black"
                            }]}
                        >
                            <Text style={[styles.tableHeader, styles.tableCell]}>{key}</Text>
                            <Text style={styles.tableCell}>{reportData.carrier[key].value}</Text>
                        </View>
                    ))
                }
            </View>

            <View style={[styles.table, styles.flexCol]}>
                <View style={[styles.flexRow, styles.noBorderBottom]}>
                    <View style={[styles.partnerTableColumn, { width: "33%", borderRight: "0.75pt solid black" }]}>
                        <Text style={[styles.tableHeader, styles.tableCell, styles.bold]}>Shipper</Text>
                        {
                            keys(reportData.shipper.basic).map((key, idx) => (
                                <View
                                    key={`shipper_basic_${idx}`}
                                    style={[styles.tableCell, {
                                        display: "flex",
                                        flexDirection: "row",
                                        justifyContent: "space-between",
                                        width: "100%",
                                        gap: 6
                                    }]}
                                >
                                    <Text style={styles.bold}>{key}:</Text>
                                    <Text style={styles.rightWrap}>{reportData.shipper.basic[key]}</Text>
                                </View>
                            ))
                        }
                    </View>
                    <View style={[styles.partnerTableColumn, { width: "33%", borderRight: "0.75pt solid black" }]}>
                        {
                            keys(reportData.shipper.details[0]).map((key, idx) => (
                                <View
                                    key={`shipper_details0_${idx}`}
                                    style={[styles.tableCell, {
                                        display: "flex",
                                        flexDirection: "row",
                                        justifyContent: "space-between",
                                        width: "100%",
                                        gap: 6
                                    }]}
                                >
                                    <Text style={styles.bold}>{key}:</Text>
                                    <Text style={styles.rightWrap}>{reportData.shipper.details[0][key]}</Text>
                                </View>
                            ))
                        }
                    </View>
                    <View style={[styles.partnerTableColumn, { width: "34%" }]}>
                        {
                            keys(reportData.shipper.details[1]).map((key, idx) => (
                                <View
                                    key={`shipper_details1_${idx}`}
                                    style={[styles.tableCell, {
                                        display: "flex",
                                        flexDirection: "row",
                                        justifyContent: "space-between",
                                        width: "100%",
                                        gap: 6
                                    }]}
                                >
                                    <Text style={styles.bold}>{key}:</Text>
                                    <Text style={styles.rightWrap}>{reportData.shipper.details[1][key]}</Text>
                                </View>
                            ))
                        }
                    </View>
                </View>
                <View style={[styles.tableCell, { borderTop: "1.5pt solid black" }]}>
                    <Text><Text style={styles.bold}>Notes:</Text> {reportData.shipper.notes}</Text>
                </View>
            </View>

            <View style={[styles.table, styles.flexCol]}>
                <View style={styles.flexRow}>
                    <View style={[styles.partnerTableColumn, { width: "33%", borderRight: "0.75pt solid black" }]}>
                        <Text style={[styles.tableHeader, styles.tableCell, styles.bold]}>Consignee</Text>
                        {
                            keys(reportData.consignee.basic).map((key, idx) => (
                                <View
                                    key={`consignee_basic_${idx}`}
                                    style={[styles.tableCell, {
                                        display: "flex",
                                        flexDirection: "row",
                                        justifyContent: "space-between",
                                        width: "100%",
                                        gap: 6
                                    }]}
                                >
                                    <Text style={styles.bold}>{key}:</Text>
                                    <Text style={styles.rightWrap}>{reportData.consignee.basic[key]}</Text>
                                </View>
                            ))
                        }
                    </View>
                    <View style={[styles.partnerTableColumn, { width: "33%", borderRight: "0.75pt solid black" }]}>
                        {
                            keys(reportData.consignee.details[0]).map((key, idx) => (
                                <View
                                    key={`consignee_details0_${idx}`}
                                    style={[styles.tableCell, {
                                        display: "flex",
                                        flexDirection: "row",
                                        justifyContent: "space-between",
                                        width: "100%",
                                        gap: 6
                                    }]}
                                >
                                    <Text style={styles.bold}>{key}:</Text>
                                    <Text style={styles.rightWrap}>{reportData.consignee.details[0][key]}</Text>
                                </View>
                            ))
                        }
                    </View>
                    <View style={[styles.partnerTableColumn, { width: "34%" }]}>
                        {
                            keys(reportData.consignee.details[1]).map((key, idx) => (
                                <View
                                    key={`consignee_details1_${idx}`}
                                    style={[styles.tableCell, {
                                        display: "flex",
                                        flexDirection: "row",
                                        justifyContent: "space-between",
                                        width: "100%",
                                        gap: 6
                                    }]}
                                >
                                    <Text style={styles.bold}>{key}:</Text>
                                    <Text style={styles.rightWrap}>{reportData.consignee.details[1][key]}</Text>
                                </View>
                            ))
                        }
                    </View>
                </View>
                <View style={[styles.tableCell, { borderTop: "1.5pt solid black" }]}>
                    <Text><Text style={styles.bold}>Notes:</Text> {reportData.consignee.notes}</Text>
                </View>
            </View>

            <View style={styles.fillInTable}>
                <View style={styles.fillInRow}>
                    <Text style={styles.fillInCell}>Accepted By</Text>
                    <Text style={styles.fillInCell}>Date</Text>
                    <Text style={[styles.fillInCell, styles.noBorderRight, { width: "50%" }]}>Signature</Text>
                </View>
                <View style={[styles.fillInRow, { borderBottom: "none" }]}>
                    <Text style={styles.fillInCell}>Driver Name</Text>
                    <Text style={styles.fillInCell}>Cell #</Text>
                    <Text style={styles.fillInCell}>Truck #</Text>
                    <Text style={[styles.fillInCell, styles.noBorderRight]}>Trailer #</Text>
                </View>
            </View>
        </Page >
    )
}

export default RateConfirmation
